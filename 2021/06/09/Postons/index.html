#include<iostream>
#include<queue>
using namespace std;
long long Now=0;    // 当前生命值
int ans=0;    // 能够捡起的数量 
int a[200004];
int n;     // position数
int main(){
    //freopen("test/div1A.txt","r",stdin);
    cin>>n;
    priority_queue<int,vector<int>,greater<int>> Q;   // 小根堆
    for(int i = 0;i<n;++i){
        cin>>a[i];
    }

    for(int i =0;i<n;++i){
        if(Now+a[i]<0){
            if(!Q.empty()&&Q.top()<a[i]){
                // 将小根堆顶部替换掉，由于Now本来就大于0，所以反悔后Now也大于0
                Now=Now+a[i]-Q.top();
                Q.pop();
                Q.push(a[i]);
                
            }else{
                continue;
            }
        }else{
            Now+=a[i];
            Q.push(a[i]);    // 把之前要的记录下来
            ans++;
        }
    }
    cout<<ans<<endl;
}
